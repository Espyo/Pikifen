first_state = idling
death_state = destroyed

script {
    idling {
        on_enter {
            get_random_int t 1 3			
			if $t = 1
				set_animation idling_1
			end_if
			if $t = 2
				set_animation idling_2
			end_if
			if $t = 3
				set_animation idling_3
			end_if
            get_random_decimal t 0 360
            turn_to_relative $t
        }
		on_pikmin_land {
            get_mob_info t trigger mob_type
			if $t != White Pikmin
				set_state destroyed
			end_if
		}
    }
    destroyed {
        on_enter {
			start_particles spore_explosion_small 0 0 0
            set_animation destroyed
        }
        on_animation_end {
			finish_dying
            delete
        }
    }
}
