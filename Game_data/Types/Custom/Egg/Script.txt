first_state = checking_health
death_state = destroyed

init {
    set_var contents random
}

script {
    checking_health {
        on_enter {
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state idling_very_broken
            end_if
            if $t < 0.66
                set_state idling_broken
            end_if
            set_state idling_healthy
        }
    }
    idling_healthy {
        on_enter {
            set_animation idling_healthy
        }
        on_damage {
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state idling_very_broken
            end_if
            if $t < 0.66
                set_state idling_broken
            end_if
        }
        on_land {
            stop
        }
    }
    idling_broken {
        on_enter {
            set_animation idling_broken
        }
        on_damage {
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state idling_very_broken
            end_if
        }
        on_land {
            stop
        }
    }
    idling_very_broken {
        on_enter {
            set_animation idling_very_broken
        }
        on_land {
            stop
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            start_dying
            start_particles egg_shard_explosion
            
            if $contents = random
                get_random_decimal c 1 100
                if $c < 55
                    set_var contents nectar
                    goto spawn
                end_if
                if $c < 90
                    set_var contents nectar_2
                    goto spawn
                end_if
                if $c < 95
                    set_var contents spicy_drop
                    goto spawn
                else
                    set_var contents bitter_drop
                    goto spawn
                end_if
            end_if
            
            label spawn
            
            if $contents = nectar
                spawn nectar
            end_if
            if $contents = nectar_2
                spawn nectar
                spawn nectar
            end_if
            if $contents = spicy_drop
                spawn spicy_drop
            end_if
            if $contents = bitter_drop
                spawn bitter_drop
            end_if
        }
        on_animation_end {
            finish_dying
            delete
        }
    }
}
