first_state = idling

init {
    set_var is_left false
}

script {
    idling {
        on_enter {
            if $is_left = true
                set_animation left_idling
            else
                set_animation right_idling
            end_if
        }
        on_receive_message {
            get_event_info m message
            if $m = destroy
                if $is_left = true
                    set_animation left_destroyed
                end_if
            end_if
        }
    }
}
