first_state = checking_health
death_state = destroyed

script {
    checking_health {
        on_enter {
            get_mob_info t self health_ratio
            if $t < 0.40
                set_state very_broken
            end_if
            if $t < 0.85
                set_state broken
            end_if
            set_state healthy
        }
    }
    healthy {
        on_enter {
            set_animation healthy
        }
        on_damage {
            get_mob_info t self health_ratio
            if $t < 0.40
                set_state very_broken_shaking
            end_if
            if $t < 0.85
                set_state broken_shaking
            end_if
            set_state healthy_shaking
        }
    }
    healthy_shaking {
        on_enter {
            set_animation healthy_shaking
        }
        on_animation_end {
            set_state healthy
        }
        on_damage {
            get_mob_info t self health_ratio
            if $t < 0.40
                set_state very_broken_shaking
            end_if
            if $t < 0.85
                set_state broken_shaking
            end_if
        }
    }
    broken {
        on_enter {
            set_animation broken
        }
        on_damage {
            get_mob_info t self health_ratio
            if $t < 0.40
                set_state very_broken_shaking
            end_if
            set_state broken_shaking
        }
    }
    broken_shaking {
        on_enter {
            set_animation broken_shaking
        }
        on_animation_end {
            set_state broken
        }
        on_damage {
            get_mob_info t self health_ratio
            if $t < 0.40
                set_state very_broken_shaking
            end_if
        }
    }
    very_broken {
        on_enter {
            set_animation very_broken
        }
        on_damage {
            set_state very_broken_shaking
        }
    }
    very_broken_shaking {
        on_enter {
            set_animation very_broken_shaking
        }
        on_animation_end {
            set_state very_broken
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            start_dying
            finish_dying
            start_particles crystal_wall_explosion 0 0 200
            set_tangible false
        }
    }
}
