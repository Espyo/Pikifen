first_state = idling

script {
    idling {
        on_enter {
            set_animation idling
            stop
            stop_particles
        }
        on_receive_message {
            get_info m message
            if $m = goal_reached
                set_state moving
            end_if
        }
    }
    
    moving {
        on_enter {
            focus link
            move_to_target focused_mob
            start_particles cardboard_box_dust -108 0 0
        }
        on_receive_message {
            get_info m message
            if $m = goal_lost
                set_state idling
            end_if
        }
        on_reach_destination {
            set_health 0
            set_state done
        }
    }
    
    done {
        on_enter {
            stop
            stop_particles
            send_message_to_links obstacle_in_place
        }
    }
}
