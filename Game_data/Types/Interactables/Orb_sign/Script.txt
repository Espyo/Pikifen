first_state = idling

script{
    idling {
        on_enter {
            set_animation idling
            set_timer 1
        }
        on_timer {
            turn_to_relative 90
            set_timer 1
        }
        on_receive_message {
            get_event_info m message
            if $m = interact
                show_message_from_var text
            end_if
        }
    }
}
