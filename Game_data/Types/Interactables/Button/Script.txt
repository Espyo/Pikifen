first_state = up

script{
    up {
        on_enter {
            set_animation up
            send_message_to_links 0
        }
        on_receive_message {
            get_event_info msg message
            if $msg = interact
                set_state down
            else
                if $msg > 0
                    set_state down
                end_if
            end_if
        }
        on_pikmin_land {
            set_state down
        }
    }

    down {
        on_enter {
            set_animation down
            send_message_to_links 1
        }
        on_animation_end {
            set_state up
        }
    }
}
