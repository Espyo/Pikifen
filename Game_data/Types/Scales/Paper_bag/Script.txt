first_state = up

init {
    set_var goal_number 20
}

script {
    up {
        on_enter {
            set_animation up
        }
        on_weight_added {
            get_mob_info t self weight
            if $t >= $goal_number
                set_state crushed
            end_if
        }
    }
    
    crushed {
        on_enter {
            set_animation crushing
            set_height 2
            set_health 0
            start_particles paper_bag_dust 0 0 200
            send_message_to_links paper_bag_crushed
        }
        on_animation_end {
            stop_particles
        }
    }
}
