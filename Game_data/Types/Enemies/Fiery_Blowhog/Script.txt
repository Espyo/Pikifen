first_state = examining
death_state = dying

init {
    set_var ignoring false
}

script {
    examining {
        on_enter {
            set_animation idling
            randomize_timer 5 7
            stop
            set_near_reach search
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state chasing
            end_if
        }
        on_timer {
            set_state exploring
        }
        on_itch {
            set_state bucking
        }
    }
    
    exploring {
        on_enter {
            set_animation moving
            randomize_timer 2.5 3.5
            move_to_target randomly
            set_near_reach search
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state chasing
            end_if
        }
        on_timer {
            set_state examining
        }
        on_far_from_home {
            set_var ignoring true
            set_state returning
        }
        on_itch {
            set_state bucking
        }
    }
    
    chasing {
        on_enter {
            set_animation moving
            move_to_target focused_mob
            set_near_reach attack
            set_far_reach chase
        }
        on_opponent_in_reach {
            set_state attacking
        }
        on_focus_off_reach {
            set_state examining
        }
        on_far_from_home {
            set_var ignoring true
            set_state returning
        }
        on_itch {
            set_state bucking
        }
    }
    
    attacking {
        on_enter {
            stop
            set_animation attacking
        }
        on_animation_end {
            set_state examining
        }
    }
    
    bucking {
        on_enter {
            stop
            set_animation bucking
        }
        on_animation_end {
            set_state examining
        }
    }
    
    returning {
        on_enter {
            move_to_target home
            set_animation moving
            set_timer 2
            set_near_reach search
        }
        on_leave {
            set_timer 0
        }
        on_timer {
            set_var ignoring false
        }
        on_reach_destination {
            set_state examining
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state chasing
            end_if
        }
    }
    
    dying {
        on_enter {
            start_dying
            set_animation dying
        }
        on_animation_end {
            finish_dying
        }
    }
    
}
