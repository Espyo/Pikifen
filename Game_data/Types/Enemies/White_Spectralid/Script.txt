first_state = resting
death_state = dying

init {
    get_mob_info home_x self x
    get_mob_info home_y self y
    get_mob_info home_z self z
}

script {
    resting_start {
        on_enter {
            set_animation resting random_time_on_spawn
            stop
            set_flying false
            set_timer 1
        }
        on_timer {
            set_state resting
        }
        on_pikmin_land {
            set_state dying
        }
    }
    
    going_home {
        on_enter {
            move_to_absolute $home_x $home_y $home_z
            set_timer 5
        }
        on_reach_destination {
            set_state resting_start
        }
        on_pikmin_land {
            set_state dying
        }
        on_timer {
            set_state flying
        }
    }
    
    resting {
        on_enter {
            set_animation resting random_time_on_spawn
            set_timer 5
            set_near_reach fear
            set_far_reach fear
        }
        on_timer {
            set_state flying
        }
        on_opponent_in_reach {
            set_state flying
        }
        on_touch_opponent {
            set_state flying
        }
        on_pikmin_land {
            set_state dying
        }
    }
    
    flying {
        on_enter {
            start_height_effect
            set_flying true
            set_animation flapping normal true
            set_timer 0.01
            set_var flutters 0
        }
        on_pikmin_land {
            set_state dying
        }
        on_timer {
            get_random_decimal x -600 600
            calculate x $home_x + $x
            get_random_decimal y -600 600
            calculate y $home_y + $y
            get_floor_z z $x $y
            calculate z $z + 100
            move_to_absolute $x $y $z

            calculate flutters $flutters + 1
            if $flutters = 20
                set_state going_home
            else
                set_timer 0.3
            end_if
        }
    }
    
    dying {
        on_enter {
            start_dying
            set_animation dying
        }
        on_animation_end {
            finish_dying
        }
    }
    
}
