first_state = sniffing
death_state = dying

init {
    set_var ignoring false
}

script {
    sniffing {
        on_enter {
            set_animation sniffing random_time_on_spawn
            stop
            set_near_reach search
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state turning
            end_if
        }
        on_itch {
            set_state shaking
        }
        on_animation_end {
            set_state exploring
        }
    }
    
    exploring {
        on_enter {
            get_random_decimal x -1000 1000
            get_random_decimal y -1000 1000
            move_to_relative $x $y
            set_animation walking
            get_random_decimal t 1.5 2.5
            set_timer $t
        }
        on_opponent_in_reach {
            focus trigger
            set_state turning
        }
        on_reach_destination {
            set_state sniffing
        }
        on_timer {
            set_state sniffing
        }
        on_far_from_home {
            set_state returning
        }
    }
    
    turning {
        on_enter {
            set_animation walking
            stop
            turn_to_target focused_mob
            set_near_reach chase
            set_far_reach close
        }
        on_opponent_in_reach {
            stop
            focus trigger
            set_state chasing
        }
        on_focus_off_reach {
            set_state sniffing
        }
        on_itch {
            set_state shaking
        }
    }
    
    chasing {
        on_enter {
            set_animation walking
            move_to_target focused_mob
            set_near_reach attack
            set_far_reach chase
        }
        on_opponent_in_reach {
            set_state attacking
        }
        on_focus_off_reach {
            set_state sniffing
        }
        on_far_from_home {
            set_var ignoring true
            set_state returning
        }
        on_itch {
            set_state shaking
        }
    }
    
    attacking {
        on_enter {
            stop
            set_animation attacking
            start_chomping 5 tongue1 tongue2
        }
        on_animation_end {
            stop_chomping
            get_mob_info t self chomped_pikmin
            if $t > 0
                set_state eating
            else
                set_state sniffing
            end_if
        }
    }
    
    eating {
        on_enter {
            set_animation chomping
        }
        on_animation_end {
            swallow_all
            set_state sniffing
        }
    }
    
    shaking {
        on_enter {
            stop
            set_animation shaking
        }
        on_animation_end {
            set_state sniffing
        }
    }
    
    returning {
        on_enter {
            move_to_target home
            set_animation walking
            set_timer 2
            set_near_reach search
        }
        on_leave {
            set_timer 0
        }
        on_timer {
            set_var ignoring false
        }
        on_reach_destination {
            set_state sniffing
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state turning
            end_if
        }
    }
    
    dying {
        on_enter {
            start_dying
            set_animation dying
        }
        on_animation_end {
            finish_dying
        }
    }
    
}
