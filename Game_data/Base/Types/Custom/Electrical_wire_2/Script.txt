first_state = safe
death_state = destroyed

global {
    on_receive_message {
        get_event_info msg message
    }
}

script {
    safe {
        on_enter {
            set_timer 1.5
            set_animation idling
            stop_particles
        }
        on_timer {
            set_state sparking
        }
        on_receive_message {
            if $msg = destroyed
                set_state destroyed
            end_if
        }
    }
    sparking {
        on_enter {
            set_timer 1.5
            start_particles electrical_wire_sparks 0 0 32
        }
        on_timer {
            set_state zapping
        }
        on_receive_message {
            if $msg = destroyed
                set_state destroyed
            end_if
        }
    }
    zapping {
        on_enter {
            set_timer 2.5
        }
        on_timer {
            set_state safe
        }
        on_receive_message {
            if $msg = destroyed
                set_state destroyed
            end_if
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            stop_particles
            start_dying
        }
    }
}
