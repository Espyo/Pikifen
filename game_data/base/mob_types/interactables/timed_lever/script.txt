first_state = off

init {
    set_var duration 5
    set_var time 0
}

global {
    on_receive_message {
        get_event_info msg message
    }
}

script{
    off {
        on_enter {
            set_animation off
            set_var time $duration
            set_health 100
            set_var msg power
            add_list_item msg $msg 0
            send_message_to_links $msg
        }
        on_receive_message {
            if $msg = interact
                set_state flipping_on
            else_if $temp = power
                get_list_item temp $msg 2
                if $temp > 0
                    set_state flipping_on
                end_if
            end_if
        }
        on_pikmin_land {
            set_state flipping_on
        }
        on_ready {
            set_var msg power
            add_list_item msg $msg 0
            send_message_to_links $msg
        }
    }

    flipping_on {
        on_enter {
            set_animation flipping_on
            set_var time $duration
            set_health 100
            play_sound click_down
            play_sound tension sound_id
        }
        on_animation_end {
            set_state on
        }
    }

    on {
        on_enter {
            set_animation on
            set_var time $duration
            set_health 100
            set_timer 0.1
            set_var msg power
            add_list_item msg $msg 1
            send_message_to_links $msg
        }
        on_receive_message {
            if $msg = interact
                set_state flipping_off
            else_if $temp = power
                get_list_item temp $msg 2
                if $temp > 0
                    set_state flipping_off
                end_if
            end_if
        }
        on_timer {
            calculate time $time - 0.1
            calculate h $time / $duration
            set_health $h
            if $time <= 0
                set_state flipping_off
            else
                set_timer 0.1
            end_if
        }
        on_pikmin_land {
            set_state flipping_off
        }
    }

    flipping_off {
        on_enter {
            set_animation flipping_off
            set_var time $duration
            set_health 100
            play_sound click_up
            stop_sound $sound_id
        }
        on_animation_end {
            set_state off
        }
    }
}
