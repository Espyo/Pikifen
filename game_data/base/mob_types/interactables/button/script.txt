first_state = up

script{
    up {
        on_enter {
            set_animation up
            set_var msg power
            add_list_item msg $msg 0
            send_message_to_links $msg
            play_sound click_up
        }
        on_receive_message {
            get_event_info msg message
            get_list_item temp $msg 1
            if $temp = interact
                set_state down
            else_if $temp = power
                get_list_item temp $msg 2
                if $temp > 0
                    set_state down
                end_if
            end_if
        }
        on_pikmin_land {
            set_state down
        }
        on_ready {
            set_var msg power
            add_list_item msg $msg 0
            send_message_to_links $msg
        }
    }

    down {
        on_enter {
            set_animation down
            set_var msg power
            add_list_item msg $msg 1
            send_message_to_links $msg
            play_sound click_down
        }
        on_animation_end {
            set_state up
        }
    }
}
