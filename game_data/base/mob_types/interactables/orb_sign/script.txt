first_state = idling

script{
    idling {
        on_enter {
            set_animation idling
            set_timer 1
        }
        on_timer {
            turn_to_relative 90
            set_timer 1
        }
        on_receive_message {
            get_event_info msg message
            if $msg = interact
                show_cutscene_message $text
            end_if
        }
    }
}
