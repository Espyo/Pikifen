init {
    set_var played_thump false
}

script {
    knocked_back {
        on_enter {
            set_var played_thump false
        }
        on_land {
            if $played_thump = false
                play_sound thump
                set_var played_thump true
            end_if
        }
    }
    ko {
        on_enter {
            set_height 20
        }
        on_leave {
            set_height 54
        }
    }
}
