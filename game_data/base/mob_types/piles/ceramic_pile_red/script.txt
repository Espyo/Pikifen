script {
    idling {
        on_damage {
            play_sound ceramic_hit
        }
        on_receive_message {
            get_event_info msg message
            if $msg = dropped_resource
                play_sound snap
                start_particles pile_sparkles_large 0 0 200
            else_if $msg = grown_resource
                start_particles pile_sparkles_large 0 0 200
            end_if
        }
    }
}
