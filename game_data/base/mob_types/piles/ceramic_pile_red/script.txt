script {
    idling {
        on_damage {
            play_sound ceramic_hit
        }
        on_receive_message {
            get_event_info m message
            if $m = dropped_resource
                play_sound snap
                start_particles pile_sparkles_large 0 0 200
            end_if
            if $m = grown_resource
                start_particles pile_sparkles_large 0 0 200
            end_if
        }
    }
}
