first_state = idling
death_state = dying

script {
    idling {
        on_enter {
			set_timer 3
            set_animation idling random_time_on_spawn
            stop
            set_near_reach search
        }
		on_timer {
			get_random_int jumps 2 5
			get_random_float x -1000 1000
			get_mob_info cur_x self x
			calculate x $x + $cur_x
            get_random_float y -1000 1000
			get_mob_info cur_y self y
			calculate y $y + $cur_y
			set_state exploring_jumping
		}
        on_opponent_in_reach {
            focus trigger
            set_state chasing_turning
        }
		on_far_from_home {
            set_state returning_resting
        }
        on_itch {
            set_state shaking
        }
		on_itch {
			set_state shaking
		}
    }
	
	exploring_jumping {
        on_enter {
            start_height_effect
            set_timer 0.3
            set_animation jumping
            set_flying true

            move_to_absolute $x $y 100
        }
        on_timer {
            set_state exploring_floating
        }
    }
    
    exploring_floating {
        on_enter {
            set_flying false
            set_gravity 0
            set_timer 0.3
            set_animation landing
        }
        on_timer {
            set_state exploring_falling
        }
    }
    
    exploring_falling {
        on_enter {
            set_gravity 1.5
        }
        on_land {
            set_state exploring_landing
        }
    }
    
    exploring_landing {
        on_enter {
			stop
            stop_height_effect
            set_gravity 1
        }
        on_animation_end {
            set_state exploring_resting
        }
    }

	exploring_resting {
        on_enter {
            stop
            set_animation idling
            
			calculate jumps $jumps - 1
			if $jumps <= 0
				set_state idling
			end_if
			
			set_timer 0.1
        }
		on_opponent_in_reach {
            focus trigger
            set_state chasing_turning
        }
        on_timer {
            set_state exploring_jumping
        }
		on_far_from_home {
            set_state returning_resting
        }
		on_itch {
			set_state shaking
		}
    }
	
    chasing_turning {
        on_enter {
            set_animation idling
            stop
            turn_to_target focused_mob
			set_near_reach attack
            set_far_reach search
			set_timer 0.1
        }
        on_opponent_in_reach {
            stop
            focus trigger
            set_state attacking
        }
		on_timer {
			set_state chasing_jumping
		}
        on_focus_off_reach {
            set_state idling
        }
        on_itch {
            set_state shaking
        }
    }
    
	attacking {
        on_enter {
            stop
            set_animation biting
            start_chomping 1 mouth
        }
        on_animation_end {
            stop_chomping
            get_mob_info t self chomped_pikmin
            if $t > 0
                set_state eating
            else
                set_state idling
            end_if
        }
    }
    
    eating {
        on_enter {
            set_animation chewing
        }
		on_frame_signal {
			swallow_all
		}
        on_animation_end {
            set_state idling
        }
        on_pikmin_land {
            set_state dying
        }
    }
	
    chasing_jumping {
        on_enter {
            start_height_effect
            set_timer 0.3
            set_animation jumping
            set_flying true
			
			get_mob_info self_x self x
			get_mob_info self_y self y
			get_mob_info focus_x focus x
			get_mob_info focus_y focus y
			
			get_angle angle $self_x $self_y $focus_x $focus_y		
			get_distance dist $self_x $self_y $focus_x $focus_y	
			calculate dist $dist - 64	
			get_coordinates_from_angle x y $angle $dist	
			
			calculate x $x + $self_x
			calculate y $y + $self_y
			
			get_mob_info z focus z
			calculate z $z + 100
			
			move_to_absolute $x $y $z
        }
        on_reach_destination {
            set_state chasing_floating
        }
        on_timer {
            set_state chasing_floating
        }
    }
    
    chasing_floating {
        on_enter {
            set_flying false
            set_gravity 0
            set_timer 0.3
            set_animation landing
        }
        on_timer {
            set_state chasing_falling
        }
    }
    
    chasing_falling {
        on_enter {
            set_gravity 1.5
        }
        on_land {
            set_state chasing_landing
        }
    }
    
    chasing_landing {
        on_enter {
			stop
            stop_height_effect
            set_gravity 1
        }
        on_animation_end {
            set_state chasing_resting
        }
    }
    
	chasing_resting {
        on_enter {
            stop
			focus trigger
            set_animation idling
            set_timer 0.1
        }
        on_timer {
            set_state chasing_turning
        }
		on_far_from_home {
            set_state returning_resting
        }
		on_itch {
			set_state shaking
		}
    }
	
	returning_jumping {
        on_enter {
            start_height_effect
            set_timer 0.3
            set_animation jumping
            set_flying true
			move_to_target home
        }
		on_reach_destination {
            set_state returning_finishing
        }
        on_timer {
            set_state returning_floating
        }
    }
    
    returning_floating {
        on_enter {
            set_flying false
            set_gravity 0
            set_timer 0.3
            set_animation landing
        }
		on_reach_destination {
            set_state returning_finishing
        }
        on_timer {
            set_state returning_falling
        }
    }
    
    returning_falling {
        on_enter {
            set_gravity 1.5
        }
		on_reach_destination {
            set_state returning_finishing
        }
        on_land {
            set_state returning_landing
        }
    }
    
    returning_landing {
        on_enter {
			stop
            stop_height_effect
            set_gravity 1
        }
		on_reach_destination {
            set_state returning_finishing
        }
        on_animation_end {
            set_state returning_resting
        }
    }

	returning_resting {
        on_enter {
            stop
            set_animation idling
            set_timer 0.1
        }
        on_timer {
            set_state returning_jumping
        }
    }
	
	returning_finishing {
        on_enter {
			stop
            stop_height_effect
            set_gravity 1
			set_animation landing
        }
        on_animation_end {
            set_state idling
        }
		on_itch {
			set_state shaking
		}
    }
	
    shaking {
        on_enter {
            set_animation shaking
        }
		on_animation_end {
            set_state idling
        }
    }
	
    dying {
        on_enter {
            set_gravity 1
            stop
            stop_vertically
            set_animation dying
            start_dying
        }
        on_animation_end {
            finish_dying
        }
    }
}
