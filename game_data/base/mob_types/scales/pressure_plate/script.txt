first_state = up

init {
    set_var goal_number 1
    set_var weight 0
}

script {
    up {
        on_enter {
            set_animation up
            play_sound click_up
            if $weight >= $goal_number
                set_state down
            end_if
            set_var msg power
            add_list_item msg $msg 0
            send_message_to_links $msg
        }
        on_weight_added {
            get_mob_info weight self weight
            if $weight >= $goal_number
                set_state down
            end_if
        }
        on_weight_removed {
            get_mob_info weight self weight
        }
        on_ready {
            set_var msg power
            add_list_item msg $msg 0
            send_message_to_links $msg
        }
    }
    
    down {
        on_enter {
            set_animation down
            play_sound click_down
            if $weight < $goal_number
                start_particles dust_pressure_plate_click 0 0 16
                set_state up
            end_if
            set_var msg power
            add_list_item msg $msg $weight
            send_message_to_links $msg
            start_particles dust_pressure_plate_click 0 0 16
        }
        on_weight_added {
            get_mob_info weight self weight
        }
        on_weight_removed {
            get_mob_info weight self weight
            if $weight < $goal_number
                start_particles dust_pressure_plate_click 0 0 16
                set_state up
            end_if
        }
    }
}
