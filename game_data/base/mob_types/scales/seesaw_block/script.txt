first_state = up

init {
    set_var height 128
    set_var pair_weight 0
    set_var start_down false
    set_var weight 0
}

global {
    on_receive_message {
        get_event_info msg message
        get_list_item temp $msg 1
        if $temp = seesaw_weight
            get_list_item pair_weight $msg 2
        end_if
    }
}

script {
    up {
        on_enter {
            if $start_down = true
                set_var start_down false
                set_state down
            end_if
            set_var height 128
            set_height 128
            set_animation up
            stop_particles
            if $weight > $pair_weight
                set_state lowering
            end_if
        }
        on_weight_added {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
            if $weight > $pair_weight
                set_state lowering
            end_if
        }
        on_weight_removed {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
        }
        on_receive_message {
            if $weight > $pair_weight
                set_state lowering
            end_if
        }
        on_ready {
            if $start_down = true
                set_var start_down false
                set_state down
            end_if
        }
    }
    
    lowering {
        on_enter {
            set_animation lowering
            start_particles dust_falling_seesaw 0 0 0
        }
        on_frame_signal {
            calculate height $height - 20
            set_height $height
        }
        on_animation_end {
            set_state down
        }
        on_weight_added {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
        }
        on_weight_removed {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
        }
    }
    
    down {
        on_enter {
            set_var height 8
            set_height 8
            set_animation down
            stop_particles
            if $weight < $pair_weight
                set_state rising
            end_if
        }
        on_weight_added {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
        }
        on_weight_removed {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
            if $weight < $pair_weight
                set_state rising
            end_if
        }
        on_receive_message {
            if $weight < $pair_weight
                set_state rising
            end_if
        }
    }
    
    rising {
        on_enter {
            set_animation rising
            start_particles dust_falling 0 0 0
        }
        on_frame_signal {
            calculate height $height + 20
            set_height $height
        }
        on_animation_end {
            set_state up
        }
        on_weight_added {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
        }
        on_weight_removed {
            get_mob_info weight self weight
            set_var msg seesaw_weight
            add_list_item msg $msg $weight
            send_message_to_links $msg
        }
    }
}
