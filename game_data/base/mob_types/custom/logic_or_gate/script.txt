first_state = idling

init {
    set_var i1 0
    set_var i2 0
    set_var i1id none
}

script {
    idling {
        on_enter {
            send_message_to_links 0
        }
        on_receive_message {
            get_event_info msg message
            get_list_item temp $msg 1

            if $temp = power
                get_list_item temp $msg 2
                get_mob_info id trigger id

                if $i1id = none
                    set_var i1id $id
                end_if
                if $id = $i1id
                    set_var i1 $temp
                else
                    set_var i2 $temp
                end_if

                calculate sum $i1 + $i2
                set_var msg power
                if $sum > 0
                    add_list_item msg $msg 1
                else
                    add_list_item msg $msg 0
                end_if
                send_message_to_links $msg
            end_if
        }
    }
}
