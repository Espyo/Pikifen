first_state = checking_health
death_state = destroyed

init {
    start_particles ice_wall_vapor
}

script {
    checking_health {
        on_enter {
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state stage_3_idling
            else_if $t < 0.66
                set_state stage_2_idling
            end_if
            set_state stage_1_idling
        }
    }
    stage_1_idling {
        on_enter {
            set_animation stage_1_idling
        }
        on_damage {
            play_sound hit
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state stage_3_shaking
            else_if $t < 0.66
                set_state stage_2_shaking
            end_if
            set_state stage_1_shaking
        }
    }
    stage_1_shaking {
        on_enter {
            set_animation stage_1_shaking
        }
        on_animation_end {
            set_state stage_1_idling
        }
        on_damage {
            play_sound hit
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state stage_3_shaking
            else_if $t < 0.66
                set_state stage_2_shaking
            end_if
        }
    }
    stage_2_idling {
        on_enter {
            set_animation stage_2_idling
        }
        on_damage {
            play_sound hit
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state stage_3_shaking
            end_if
            set_state stage_2_shaking
        }
    }
    stage_2_shaking {
        on_enter {
            set_animation stage_2_shaking
        }
        on_animation_end {
            set_state stage_2_idling
        }
        on_damage {
            play_sound hit
            get_mob_info t self health_ratio
            if $t < 0.33
                set_state stage_3_shaking
            end_if
        }
    }
    stage_3_idling {
        on_enter {
            set_animation stage_3_idling
        }
        on_damage {
            play_sound hit
            set_state stage_3_shaking
        }
    }
    stage_3_shaking {
        on_enter {
            set_animation stage_3_shaking
        }
        on_animation_end {
            set_state stage_3_idling
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            start_dying
            finish_dying
            stop_particles
            start_particles enemy_defeat 0 0 129
            start_particles ice_wall_explosion 0 0 129
            play_sound destroyed
            play_sound obstacle_cleared
            set_tangible false
            set_shadow_visibility false
            //Set height to 1 so the debris show below the leaders.
            set_height 1
        }
    }
}
