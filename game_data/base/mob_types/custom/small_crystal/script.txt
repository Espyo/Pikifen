first_state = capturing
death_state = destroying

init {
    set_var capturing true
}

script {
    capturing {
        on_enter {
            set_timer 0.1
        }
        on_timer {
            set_var capturing false
            set_state stage_1_idling
        }
        on_touch_object {
            if $capturing = true
                focus trigger
                hold_focused_mob crystal
                set_var capturing false
            end_if
        }
    }

    stage_1_idling {
        on_enter {
            set_animation stage_1_idling
        }
        on_damage {
            play_sound hit
            get_mob_info t self health_ratio
            if $t < 0.85
                set_state stage_2_shaking
            end_if
            set_state stage_1_shaking
        }
    }

    stage_1_shaking {
        on_enter {
            set_animation stage_1_shaking
        }
        on_animation_end {
            set_state stage_1_idling
        }
        on_damage {
            play_sound hit
            get_mob_info t self health_ratio
            if $t < 0.85
                set_state stage_2_shaking
            end_if
        }
    }

    stage_2_idling {
        on_enter {
            set_animation stage_2_idling
        }
        on_damage {
            play_sound hit
            set_state stage_2_shaking
        }
    }

    stage_2_shaking {
        on_enter {
            set_animation stage_2_shaking
        }
        on_animation_end {
            set_state stage_2_idling
        }
        on_damage {
            play_sound hit
        }
    }

    destroying {
        on_enter {
            set_tangible false
            start_particles crystal_explosion_small 0 0 100
            set_animation destroyed
            play_sound destroyed
        }
        on_animation_end {
            throw_focused_mob 0 0 0 0
            delete
        }
    }
}
