first_state = safe
death_state = destroyed

init {
    teleport_to_relative 0 -80 0
}

script {
    safe {
        on_enter {
            set_timer 1.5
            set_animation safe
            stop_particles
        }
        on_timer {
            set_state sparking
        }
    }
    sparking {
        on_enter {
            set_timer 1.5
            start_particles electrical_wire_sparks 0 0 32
            play_sound sparks sound_id
        }
        on_leave {
            stop_sound $sound_id
        }
        on_timer {
            set_state zapping
        }
    }
    zapping {
        on_enter {
            set_timer 2.5
            set_animation zapping
            play_sound electricity sound_id
        }
        on_leave {
            stop_sound $sound_id
        }
        on_timer {
            set_state safe
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            stop_particles
            start_particles enemy_death 0 0 25
            send_message_to_links destroyed
            start_dying
        }
    }
}
