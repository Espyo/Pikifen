first_state = shooting
death_state = destroyed

script{
    shooting {
        on_enter {
            set_animation shooting
            start_particles gas_pipe 0 0 24
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            stop_particles
            start_particles enemy_death 0 0 150
            start_dying
        }
    }
}
