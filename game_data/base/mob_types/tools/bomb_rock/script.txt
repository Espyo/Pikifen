first_state = idling
death_state = exploding

init {
    //This gets set by the engine when the Pikmin releases it gently (e.g. entering an Onion).
    set_var gentle_release false
    set_var to_delete false
}

script {
    idling {
        on_enter {
            set_animation idling random_time_on_spawn
            set_holdable pikmin enemies
        }
        on_damage {
            set_state primed
        }
        on_held {
            get_mob_info h trigger mob_category
            if $h = pikmin
                set_state held_by_pikmin
            end_if
        }
        on_hitbox_touch_eat {
            get_chomped
            set_state held_by_enemy
        }
        on_land {
            stop
        }
    }
    
    held_by_pikmin {
        on_enter {
            set_animation held
        }
        on_released {
            if $gentle_release = true
                set_state idling
            else
                set_state primed
            end_if
        }
    }
    
    held_by_enemy {
        on_enter {
            set_animation held
            remove_status bomb_rock_health_drain
            stop_sound $sound_id
        }
        on_released {
            set_state primed
        }
        on_swallowed {
            stop_sound $sound_id
            set_var to_delete true
            delete
        }
    }
    
    primed {
        on_enter {
            if $to_delete = false
                set_animation primed
                set_holdable enemies
                receive_status bomb_rock_health_drain
                play_sound wick sound_id
                stop
            end_if
        }
        on_hitbox_touch_eat {
            get_chomped
            set_state held_by_enemy
        }
        on_land {
            stop
        }
    }
    
    exploding {
        on_enter {
            spawn explosion
            stop_sound $sound_id
            set_var to_delete true
            delete
        }
    }
}
